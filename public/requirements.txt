import os
from groq import Groq

# سحب المفتاح من البيئة (سواء Vercel أو جهازك)
client = Groq(api_key=os.environ.get("GROQ_API_KEY"))

def get_ai_response(user_input):
    chat_completion = client.chat.completions.create(
        messages=[
            {
                "role": "system",
                "content": "You are Neura, a compassionate AI mental health support assistant. Analyze the user's emotion and respond with empathy."
            },
            {
                "role": "user",
                "content": user_input,
            }
        ],
        model="llama3-8b-8192", # موديل سريع جداً ومناسب للـ Emotion Analysis
    )
    return chat_completion.choices[0].message.content
